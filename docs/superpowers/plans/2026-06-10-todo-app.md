> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 단일 `index.html` 파일로 추가·완료·수정·삭제·필터 기능을 갖춘 인메모리 TODO 앱을 구현한다.

### Task 2: 인메모리 상태 + render() 뼈대

**Files:**
- Modify: `todo/index.html` — `<script>` 블록

- [ ] **Step 1: 상태와 render() 뼈대 작성**

`<script>` 내용을 아래로 교체한다.

```js
let todos = [];
let filter = 'all';
let nextId = 1;

function render() {
  const list = document.getElementById('todo-list');
  const summary = document.getElementById('summary');

  const visible = todos.filter(t => {
    if (filter === 'active')    return !t.done;
    if (filter === 'completed') return  t.done;
    return true;
  });

  if (visible.length === 0) {
    list.innerHTML = `<li class="text-sm text-gray-400 text-center py-4">
      ${todos.length === 0 ? '할 일이 없습니다.' : '해당하는 항목이 없습니다.'}
    </li>`;
  } else {
    list.innerHTML = visible.map(t => renderItem(t)).join('');
  }

  const total = todos.length;
  const done  = todos.filter(t => t.done).length;
  summary.textContent = total > 0 ? `총 ${total}개 · 미완료 ${total - done}개` : '';
}

function renderItem(t) {
  return `
    <li class="flex items-center gap-2 p-2 rounded-lg hover:bg-gray-50 group" data-id="${t.id}">
      <input type="checkbox" ${t.done ? 'checked' : ''}
        onchange="toggleTodo(${t.id})"
        class="w-4 h-4 cursor-pointer accent-blue-500" />
      <span class="flex-1 text-sm ${t.done ? 'line-through text-gray-400' : 'text-gray-700'}">${escapeHtml(t.text)}</span>
      <button onclick="startEdit(${t.id})"
        class="text-xs text-blue-400 hover:text-blue-600 opacity-0 group-hover:opacity-100 transition">수정</button>
      <button onclick="deleteTodo(${t.id})"
        class="text-xs text-red-400 hover:text-red-600 opacity-0 group-hover:opacity-100 transition">삭제</button>
    </li>`;
}

function escapeHtml(str) {
  return str.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}

render();
```

```js
function toggleTodo(id) {
  const todo = todos.find(t => t.id === id);
  if (!todo) return;
  todo.done = !todo.done;
  render();
}

function deleteTodo(id) {
  todos = todos.filter(t => t.id !== id);
  render();
}
```

- [ ] **Step 2: startEdit() / saveEdit() / handleEditKey() 함수 추가**

`deleteTodo()` 뒤에 추가한다.

```js
function startEdit(id) {
  todos = todos.map(t => ({ ...t, editing: t.id === id }));
  render();
  const input = document.getElementById(`edit-input-${id}`);
  if (input) { input.focus(); input.select(); }
}

function saveEdit(id, value) {
  const todo = todos.find(t => t.id === id);
  if (!todo) return;
  const trimmed = value.trim();
  todo.text    = trimmed || todo.text; // 빈 값이면 원래 텍스트 유지
  todo.editing = false;
  render();
}

function handleEditKey(e, id) {
  if (e.key === 'Enter') {
    e.target.blur(); // blur 이벤트가 saveEdit() 호출
  }
  if (e.key === 'Escape') {
    const todo = todos.find(t => t.id === id);
    if (todo) { todo.editing = false; render(); }
  }
}
```

document.getElementById('filter-tabs').addEventListener('click', e => {
  const btn = e.target.closest('[data-filter]');
  if (!btn) return;
  filter = btn.dataset.filter;
  updateFilterTabs();
  render();
});

function updateFilterTabs() {
  document.querySelectorAll('.filter-btn').forEach(btn => {
    const active = btn.dataset.filter === filter;
    btn.className = `filter-btn px-3 py-1 rounded-full text-sm transition ${
      active ? 'bg-blue-500 text-white' : 'bg-gray-100 text-gray-600 hover:bg-gray-200'
    }`;
  });
}
